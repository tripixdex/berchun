\subsubsection*{Исходные данные}
\begin{itemize}
    \item Tc = 14 с
    \item Ts = 65 с
    \item λ = 0.071429 1/с
    \item μ = 0.015385 1/с
\end{itemize}
\subsubsection*{Расчётная схема}

\subsection*{1.2. Система с ограниченной очередью}
Рассматривается система с конечным числом мест в очереди и потерей заявки при полном заполнении системы.

\subsubsection*{Исходные данные}
\begin{itemize}
    \item Tc = 14 с
    \item Ts = 65 с
    \item λ = 0.071429 1/с
    \item μ = 0.015385 1/с
\end{itemize}
\subsubsection*{Расчётная схема}
\begin{figure}[H]
\centering
\includegraphics[width=0.82\textwidth]{\detokenize{assets/task1_2__scheme.png}}
\caption*{Рисунок 1.2.1. Расчётная схема.}
\end{figure}
\subsubsection*{Формулы вероятностей состояний}
\[
a = \lambda / \mu,\quad \rho_n = \lambda /(n\mu)
\]

\[
p_0 = \left(\sum_{k=0}^{n}\frac{a^k}{k!} + \frac{a^n}{n!}\sum_{r=1}^{m}\rho_n^{\,r}\right)^{-1}
\]

\[
p_k = \frac{a^k}{k!}p_0,\quad k=0,\ldots,n
\]

\[
p_{n+r} = \frac{a^n}{n!}\rho_n^{\,r}p_0,\quad r=1,\ldots,m
\]
\subsubsection*{Формулы производных метрик}
\[
P_{\mathrm{отк}} = p_{n+m}
\]

\[
M_{\mathrm{зан}} = \sum_{k=0}^{n+m}\min(k,n)\,p_k
\]

\[
P_{\mathrm{оч}} = \sum_{k=n+1}^{n+m}p_k
\]

\[
L_{\mathrm{оч}} = \sum_{k=n+1}^{n+m}(k-n)\,p_k,\quad K_{\mathrm{мест}} = L_{\mathrm{оч}}/m
\]
\subsubsection*{Результаты и пояснения}
Построены оба требуемых sweep-режима: по числу мест в очереди при фиксированном числе операторов и по числу операторов при фиксированном числе мест в очереди.

Во всех сериях использованы диапазоны по операторам 1..15 и по местам в очереди 1..15 без ручного отбора точек.

\subsection*{1.3. Система с неограниченной очередью}
Рассматривается система с бесконечной очередью; стационарные метрики допустимы только при условии λ < nμ.

\subsubsection*{Исходные данные}
\begin{itemize}
    \item Tc = 14 с
    \item Ts = 65 с
    \item λ = 0.071429 1/с
    \item μ = 0.015385 1/с
\end{itemize}
\subsubsection*{Расчётная схема}
\begin{figure}[H]
\centering
\includegraphics[width=0.82\textwidth]{\detokenize{assets/task1_3__scheme.png}}
\caption*{Рисунок 1.3.1. Расчётная схема.}
\end{figure}
\subsubsection*{Формулы вероятностей состояний}
\[
a = \lambda / \mu,\quad \rho_n = \lambda /(n\mu)
\]

\[
p_0 = \left(\sum_{k=0}^{n-1}\frac{a^k}{k!} + \frac{a^n}{n!(1-\rho_n)}\right)^{-1},\quad \rho_n < 1
\]

\[
P_{\mathrm{wait}} = \frac{a^n}{n!(1-\rho_n)}p_0
\]
\subsubsection*{Формулы производных метрик}
\[
M_{\mathrm{зан}} = a,\quad K_{\mathrm{загр}} = a/n
\]

\[
P_{\mathrm{оч}} = P_{\mathrm{wait}}\rho_n
\]

\[
L_{\mathrm{оч}} = \frac{P_{\mathrm{wait}}\rho_n}{1-\rho_n}
\]
\subsubsection*{Результаты и пояснения}
Нестационарными остаются режимы n = 1, 2, 3, 4; для них стационарные метрики в отчёте не подставляются.

    \item Tc = 14 с

\subsection*{2.1. Метрики производственного участка по числу наладчиков}
Рассматривается конечный источник заявок: состояние i равно числу неисправных станков.

\subsubsection*{Исходные данные}
\begin{itemize}
    \item N = 36
    \item Tc = 104 мин
    \item Ts = 50 мин
    \item λ = 0.009615 1/мин
    \item μ = 0.02 1/мин
\end{itemize}
\subsubsection*{Расчётная схема}
\begin{figure}[H]
\centering
\includegraphics[width=0.82\textwidth]{\detokenize{assets/task2_1__scheme.png}}
\caption*{Рисунок 2.1.1. Расчётная схема.}
\end{figure}
\subsubsection*{Формулы вероятностей состояний}
\[
\lambda_i = (N-i)\lambda,\quad \mu_i = \min(i,r)\mu
\]

\[
p_i = p_{i-1}\lambda_{i-1}/\mu_i,\quad i=1,\ldots,N
\]

\[
p_0 = \left(1 + \sum_{i=1}^{N}\prod_{j=1}^{i}\lambda_{j-1}/\mu_j\right)^{-1}
\]
\subsubsection*{Формулы производных метрик}
\[
M_{\mathrm{пр}} = \sum_{i=0}^{N}i\,p_i,\quad M_{\mathrm{ож}} = \sum_{i=0}^{N}\max(i-r,0)\,p_i
\]

\[
M_{\mathrm{зан}} = \sum_{i=0}^{N}\min(i,r)\,p_i,\quad K_{\mathrm{загр}} = M_{\mathrm{зан}}/r
\]

\[
P_{\mathrm{ож}} = \frac{\sum_{i=r}^{N-1}(N-i)p_i}{\sum_{i=0}^{N-1}(N-i)p_i}
\]
\subsubsection*{Результаты и пояснения}
Вероятность ожидания в задаче 2.1 трактуется как arrival-weighted вероятность ожидания для нового отказа.

На концах диапазона r = 1 и r = 36 значения P\_ож составляют 1 и 0 соответственно.
